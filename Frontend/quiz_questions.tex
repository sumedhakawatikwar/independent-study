\section*{Quiz}

\begin{enumerate}
  \item Who sang the title song for the latest Bond film, No Time to Die?
  \begin{enumerate}[label=(\alph*)]
    \item Adele
    \item Sam Smith
    \item Billie Eilish
    \item Adyen
  \end{enumerate}
  \textbf{Answer:} Billie Eilish

  \item Which flies a green, white, and orange (in that order) tricolor flag?
  \begin{enumerate}[label=(\alph*)]
    \item Ireland
    \item Ivory Coast
    \item Italy
    \item Germany
  \end{enumerate}
  \textbf{Answer:} Ireland

  \item What company makes the Xperia model of smartphone?
  \begin{enumerate}[label=(\alph*)]
    \item Samsung
    \item Sony
    \item Nokia
    \item Apple
  \end{enumerate}
  \textbf{Answer:} Sony

  \item Which city is home to the Brandenburg Gate?
  \begin{enumerate}[label=(\alph*)]
    \item Vienna
    \item Zurich
    \item Berlin
    \item Hyderabad
  \end{enumerate}
  \textbf{Answer:} Berlin

\end{enumerate}
